-- Таблица: utm.ReportImported
-- Комментарий: Отчеты (сведения) об импорте алкогольной продукции
-- Внешние ключи:
--   OwnerId → nsi.NSI_Clients/Owner_ID
--   ImporterOwnerid → nsi.NSI_Clients/Owner_ID
--   SupplierOwnerid → nsi.NSI_Clients/Owner_ID
--   Country_Code → nsi.ADDR_COUNTRY/country_code
--   Alc_Code → nsi.NSI_Products_nc/Alc_Code
--   Region_Code → nsi.ADDR_REGION/Region_Code
CREATE TABLE utm.ReportImported (
    -- Уникальный идентификатор позиции в отчете об импорте (суррогатный идентификатор)
    PosId INT,
    -- Внутренний идентификатор записи (суррогатный идентификатор)
    iid INT,
    -- Дата и время изменения записи в базе
    ChangeDate TIMESTAMP(3),
    -- Регистрациоонный идентификатор документа импорта в системе ЕГАИС
    DocId VARCHAR(50),
    -- Код организации (СИО) - отправителя документа в систему ЕГАИС
    OwnerId VARCHAR(50),
    -- Бухгалтерский номер документа импорта.
    DocNumber VARCHAR(100),
    -- Бухгалтерская дата документа импорта.
    DocDate DATE,
    -- Дата фактического ввоза продукции.
    ImportedDate DATE,
    -- Код организации - импортера.
    ImporterOwnerid VARCHAR(50),
    -- Код организации - поставщика.
    SupplierOwnerid VARCHAR(50),
    -- Номер грузовой таможенной декларации (GTD).
    GTDNUMBER VARCHAR(100),
    -- Дата грузовой таможенной декларации.
    GTDDate DATE,
    -- Номер контракта на поставку.
    ContractNUMBER VARCHAR(100),
    -- Дата контракта.
    ContractDate DATE,
    -- Код страны происхождения.
    Country_Code VARCHAR(3),
    -- Код алкогольной продукции в справочнике ЕГАИС (алкокод)
    Alc_Code INT,
    -- Количество продукции для фасованной продукции в штуках, для нефасованной в литрах
    Quantity NUMERIC(18,4),
    -- Партия продукции (номер партии).
    Party VARCHAR(100),
    -- Регистрационный номер формы 1 (для учета алкоголя, также называют справка 1 или справка А).
    F1RegId VARCHAR(50),
    -- Фактический Процент алкоголя в продукции
    AlcPercent NUMERIC(18,3),
    -- ИНН организации импортера 
    INN VARCHAR(12),
    -- КПП организации импортера по месту осуществления деятельности
    KPP VARCHAR(9),
    -- Код региона организации импортера 
    Region_Code VARCHAR(2),
    -- Тип продукции  АП:  ССНП - Спиртосодержащая непищева продукция; ССП - Спиртосодержащая продукция; ЭС  - Спирты; ССПП Спиртосодержащая пищева продукция; АП - Алкогольная продукция
    ProductVid_Code VARCHAR(10),
    -- Признак того, что продукция не расфасована
    isUnpacked BOOLEAN,
    -- Ёмкость единицы продукции (литры).
    Capacity NUMERIC(18,4),
    -- Алкогольная крепость продукции из справочника (расчетная, использовать только если не заполнена фактическая)
    AlcPerc NUMERIC(18,3),
    -- признак что продукция не маркируется ФСМ
    UNFSM_FLAG BOOLEAN,
    -- Идентификатор склада временного хранения (бонди).
    BONDEDWAREHOUSE VARCHAR(50)
);

    PosIdentity VARCHAR(50)

-- Таблица: data_showcase.WayBill
-- Комментарий: Информация о транспортных накладных (оптовом обороте) алкогольной продукции, каждая строка — это позиция в накладной, информация по шапке накладной дублируется.
-- Внешние ключи:
--   Client_id → nsi.NSI_Clients/Owner_ID
--   Shipper → nsi.NSI_Clients/Owner_ID
--   Consignee → nsi.NSI_Clients/Owner_ID
--   ProductCode → nsi.NSI_Products_nc/Alc_Code
CREATE TABLE data_showcase.WayBill (
    -- Идентификатор позиции накладной
    wbpid INT NOT NULL,
    -- Идентификатор накладной, он же номер фиксации или регистрационный номер
    wbid VARCHAR(50) NOT NULL,
    -- Суррогатный ключ записи
    iid INT,
    -- СИО или код организации — отправителя документа, он же овнер или овнерид
    Client_id VARCHAR(50),
    -- Номер накладной
    "Number" VARCHAR(200),
    -- Дата накладной
    Date DATE,
    -- Тип накладной (WBReturnFromMe - возвратная, WBInvoiceFromMe - поставка)
    Type VARCHAR(50),
    -- СИО или код организации — отправителя продукции, он же овнер или овнерид
    Shipper VARCHAR(50),
    -- СИО или код организации — получателя продукции, он же овнер или овнерид
    Consignee VARCHAR(50),
    -- Фактическая дата отгрузки продукции
    ShippingDate DATE,
    -- Дата вставки записи в таблицу
    InsertDate TIMESTAMP(3) NOT NULL,
    -- Транспортная компания
    TRAN_COMPANY VARCHAR(510),
    -- Юридический номер фиксации
    EGAISFixNumber VARCHAR(50),
    -- Юридическая дата фиксации
    EGAISFixDate DATE,
    -- Тип документа из файла (например, WayBill_v2, WayBill_v3 и т.п.)
    fileDocType VARCHAR(50),
    -- Статус накладной: Confirmed — зафиксирована, Receipted — получена (остальные статусы означают отказ в фиксации)
    StateAlias VARCHAR(50),
    -- Дата последнего изменения данных в базе
    ChangeDate TIMESTAMP(3),
    -- Номер акта подтверждения прихода получателем
    ActNumber VARCHAR(100),
    -- Дата акта подтверждения прихода получателем
    ActDate DATE,
    -- Код продукции ЕГАИС, он же алкокод
    ProductCode INT,
    -- Количество продукции в позиции: для нефасованной продукции — в литрах
    Quantity NUMERIC(18,4),
    -- Стоимость продукции за единицу
    Price NUMERIC(18,4),
    -- Крепость продукции (% об.)
    AlcPercent NUMERIC(18,3),
    -- Реальное количество продукции с учетом акта приемки получателя
    RealQuantity NUMERIC(18,4),
    -- ИНН отправителя
    Shipper_INN VARCHAR(12),
    -- КПП отправителя по месту нахождения
    Shipper_KPP VARCHAR(9),
    -- Код региона РФ по месту нахождения отправителя
    Shipper_Region_Code VARCHAR(2),
    -- ИНН получателя
    Consignee_INN VARCHAR(12),
    -- КПП получателя по месту нахождения
    Consignee_KPP VARCHAR(9),
    -- Код страны получателя (для поставок на экспорт код не равен 643)
    Consignee_Country_Code VARCHAR(3),
    -- Код региона РФ получателя
    Consignee_Region_Code VARCHAR(2),
    -- Дата розлива продукции
    BottlingDate DATE,
    -- Объем продукции в даллах (декалитрах), заранее подсчитанный (1 дал = 10 литров)
    Volume_dal NUMERIC(18,4),
    -- Количество бутылок для фасованной алкогольной продукции
    Count_btl NUMERIC(18,0),
    -- Заказчик транспортировки (может отличаться от отправителя)
    TRAN_CUSTOMER VARCHAR(510),
    -- Водитель автомобиля
    TRAN_DRIVER VARCHAR(510),
    -- Тип автомобиля
    TRAN_TYPE VARCHAR(100)
);

-- Таблица: egais_or_cheque.production_chq_btl
-- Комментарий: Данные по розничным продажам через кассу алкогольной продукции, учет продажи происходит в разрезе штрих-кодов (идентификаторов бутылки) поэтому каждая строка содержит одну бутылку.
-- Внешние ключи:
--   ownerId → nsi.NSI_Clients/Owner_ID
--   alc_code → nsi.NSI_Products_nc/Alc_Code
CREATE TABLE egais_or_cheque.production_chq_btl (
    -- Штрихкод или  идиентификатор бутылки (также идентификатор егаис) уникальный для каждой бутылки
    bcode VARCHAR(255),
    -- Цена единицы продукции в рублях
    price NUMERIC(18,2),
    -- Признак возврата товара (true — возврат, false — продажа).
    vozvrat BOOLEAN,
    -- Код EAN/UPC товара.
    ean VARCHAR(14),
    -- Номер смены кассира.
    shift INT,
    -- Номер чека либо номер фискального документа
    number INT,
    -- Дата чека либо фискального документа
    dt TIMESTAMP,
    -- Идентификатор владельца (участника оборота алкоголя).
    ownerId VARCHAR(12),
    -- Уникальный идентификатор транзакции (например, UUID).
    transport_id VARCHAR(36),
    -- Номер фискального накопителя либо номер кассы
    kassa VARCHAR(100),
    -- Код алкогольной продукции в справочнике ЕГАИС (алкокод)
    alc_code INT,
    -- Регистрационный номер формы 2 (для учета алкоголя, также называют справка 2 или справка Б). По факту идентификатор партии на складе
    form2 VARCHAR(50),
    -- Номер счёта или кассового чека в случае продажи юридическому лицу по накладной
    billnumber VARCHAR(50),
    -- Номер транспортной накладной в случае продажи юридическому лицу по накладной
    ttnnumber VARCHAR(50),
    -- Количество проданной/возвращённой продукции. 
    quantity NUMERIC(18,3),
    -- Признак пивной продукции (true — пиво или пивной напиток). Для пива не указывается штрих код
    beer BOOLEAN,
    -- Дата и время вставки записи в базу.
    insertdate TIMESTAMP
);

-- Таблица: nsi.ADDR_COUNTRY
-- Комментарий: Справочник стран

    id INT,

    full_name VARCHAR(128),

-- Таблица: nsi.ADDR_REGION
-- Комментарий: Справочник регионов (Субъектов) РФ
-- Внешние ключи:
--   country_code → nsi.ADDR_COUNTRY/country_code
CREATE TABLE nsi.ADDR_REGION (
    -- Код субъекта Российской Федерации (например, "77" — Москва).
    region_code VARCHAR(2),
    -- Код страны, к которой относится регион (для РФ 643).
    country_code VARCHAR(3),
    -- Наименование региона (субъекта Российской Федерации)
    name VARCHAR(80),
    -- Сокращённое наименование региона (например, "г", "обл", "респ").
    socr VARCHAR(20),
    -- Код региона по классификатору ОКАТО/ОКТМО.
    code VARCHAR(13),
    -- Почтовый индекс региона (основной или центральный).
    index VARCHAR(6),
    -- Код ГНИНМБ (Государственного номера налогового инспекторского участка).
    gninmb VARCHAR(4)
);

-- Таблица: nsi.NSI_Products_nc
-- Комментарий: Справочник алкогольной продукции ЕГАИС
-- Внешние ключи:
--   Owner_ID → nsi.NSI_Clients/Owner_ID
--   Country_Code → nsi.ADDR_COUNTRY/country_code
--   Importer_OwnerID → nsi.NSI_Clients/Owner_ID

--   PACKAGE_TYPE_CODE → nsi.Product_PackagingType/PackageType

CREATE TABLE nsi.NSI_Products_nc (
    -- Код алкогольной продукции в справочнике ЕГАИС (алкокод)
    Alc_Code INT,

    -- Идентификатор производителя  продукции в системе ЕГАИС.
    Owner_ID VARCHAR(50),
    -- Объём единицы продукции (в литрах).
    Capacity NUMERIC(18,4),
    -- Фактическая алкогольная крепость (% об.).
    AlcPerc NUMERIC(18,3),
    -- Минимально допустимая крепость (% об.).
    AlcPercMin NUMERIC(18,3),
    -- Максимально допустимая крепость (% об.).
    AlcPercMax NUMERIC(18,3),
    -- Код вида продукции
    ProductType_Code VARCHAR(5),
    -- Тип (категория) продукции  АП:  ССНП - Спиртосодержащая непищева продукция; ССП - Спиртосодержащая продукция; ЭС  - Спирты; ССПП Спиртосодержащая пищева продукция; АП - Алкогольная продукция
    ProductVid_Code VARCHAR(10),
    -- Код страны происхождения продукции.
    Country_Code VARCHAR(3),
    -- Признак что продукция не фасованная
    isUnpacked BOOLEAN,
    -- Идентификатор импортера  продукции в системе ЕГАИС.
    Importer_OwnerID VARCHAR(50),
    -- Наименование бренда продукции.(не всегда заполнен)
    BRAND VARCHAR(510),
    -- Код типа упаковки
    PACKAGE_TYPE_CODE VARCHAR(5),
    -- Срок годности продукции (в днях).
    shelflife INT
);
